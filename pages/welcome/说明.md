默认使用hash模式，要想改成history模式，必须有服务器支持，在服务器端配置。

uni.navigateTo‌：保留当前页面，跳转到应用内的某个页面
‌uni.redirectTo‌：关闭当前页面，跳转到应用内的某个页面
‌uni.reLaunch‌：关闭所有页面，打开到应用内的某个页面


navigateTo不能跳转tabbar页面，切换tabbar页面要用switchTab。

<script setup>
	const addUserViaCloudFunction = async () => {
		try {
			const result = await uniCloud.callFunction({
				name: 'handle_data',
				data: {
					action: 'add',
					category:'pet',
					desc:'图片描述文字',
					comments: [
						{
							user_img : '',
							user_name : 'testname111',
							comMsg : '评论的信息2222',
							time : '2020-02-30'
						}
					]
				}
			})

			console.log('云函数返回:', result)

			if (result.result.code === 200) {
				uni.showToast({
					title: '添加成功',
					icon: 'success'
				})
			}
		} catch (error) {
			console.error('调用云函数失败:', error)
		}
	}
</script>
